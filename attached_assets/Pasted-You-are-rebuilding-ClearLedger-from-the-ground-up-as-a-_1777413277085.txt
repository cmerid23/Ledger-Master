  validateEntry(lines: JournalLine[]) {
    const totalDebits = lines.reduce((sum, l) => sum + l.debit, 0)

    const debits = await sumDebits(accountId, asOfDate)
    const credits = await sumCredits(accountId, asOfDate)
    
    if (account.normal_balance === 'debit') {

    } else {
      return credits - debits  // Liabilities, Equity, Income
    }
  }

    // Apply suggestions back to transactions
    return transactions.map((t, i) => {
      const suggestion = suggestions.find(s => s.index === i)
      if (suggestion) {
        const account = chartOfAccounts.find(a => a.code === suggestion.account_code)
        return { ...t, suggested_account_id: account?.id, suggested_confidence: suggestion.confidence }
      }
      return t
    })
  }
}

class ReportsEngine {

  async getProfitAndLoss(businessId: string, from: string, to: string) {
    
    // Get all journal_lines within date range
    // Group by account type using the accounts table
    // Calculate net balance for each account
    
    const incomeAccounts = await getAccountBalances(businessId, 'income', from, to)
    const cogsAccounts = await getAccountBalances(businessId, 'cogs', from, to)
    const expenseAccounts = await getAccountBalances(businessId, 'expense', from, to)
    
    const totalIncome = sum(incomeAccounts)
    const totalCOGS = sum(cogsAccounts)
    const grossProfit = totalIncome - totalCOGS
    const totalExpenses = sum(expenseAccounts)
    const netProfit = grossProfit - totalExpenses
    
    return {
      sections: [
        {
          title: 'Income',
          accounts: incomeAccounts,
          total: totalIncome
        },
        {
          title: 'Cost of Goods Sold',
          accounts: cogsAccounts,
          total: totalCOGS
        },
        {
          title: 'Gross Profit',
          total: grossProfit
        },
        {
          title: 'Operating Expenses',
          accounts: expenseAccounts,
          total: totalExpenses
        },
        {
          title: 'Net Profit',
          total: netProfit,
          highlight: true
        }
      ],
      summary: { totalIncome, totalCOGS, grossProfit, totalExpenses, netProfit }
    }
  }

  async getBalanceSheet(businessId: string, asOfDate: string) {
    
    const assetAccounts = await getAccountBalances(businessId, 'asset', null, asOfDate)
    const liabilityAccounts = await getAccountBalances(businessId, 'liability', null, asOfDate)
    const equityAccounts = await getAccountBalances(businessId, 'equity', null, asOfDate)
    
    // Retained earnings = cumulative net profit from inception to asOfDate
    const retainedEarnings = await getCumulativeNetProfit(businessId, asOfDate)
    
    const totalAssets = sum(assetAccounts)
    const totalLiabilities = sum(liabilityAccounts)
    const totalEquity = sum(equityAccounts) + retainedEarnings
    
    // CRITICAL VALIDATION:
    if (Math.abs(totalAssets - (totalLiabilities + totalEquity)) > 0.01) {
      console.error('BALANCE SHEET OUT OF BALANCE', { totalAssets, totalLiabilities, totalEquity })
    }
    
    return {
      assets: { accounts: assetAccounts, total: totalAssets },
      liabilities: { accounts: liabilityAccounts, total: totalLiabilities },
      equity: { accounts: equityAccounts, retainedEarnings, total: totalEquity },
      balanced: Math.abs(totalAssets - (totalLiabilities + totalEquity)) < 0.01
    }
  }

  async getCashFlow(businessId: string, from: string, to: string) {
    
    // Operating: changes in bank accounts from operating activities
    // Get all bank account transactions categorized as income/expense
    
    const bankAccounts = await getBankAccounts(businessId)
    
    // Operating activities
    const cashReceipts = await sumDebits(bankAccounts, from, to)  // money in
    const cashPayments = await sumCredits(bankAccounts, from, to)  // money out
    const netOperating = cashReceipts - cashPayments
    
    return {
      operating: { receipts: cashReceipts, payments: cashPayments, net: netOperating },
      investing: { net: 0 },   // for later
      financing: { net: 0 },   // for later
      netChange: netOperating,
      openingBalance: await getBankBalance(businessId, from),
      closingBalance: await getBankBalance(businessId, to)
    }
  }

  async getAccountsReceivableAging(businessId: string, asOfDate: string) {
    
    const unpaidInvoices = await getUnpaidInvoices(businessId, asOfDate)
    
    return unpaidInvoices.reduce((aging, invoice) => {
      const daysPastDue = daysBetween(invoice.due_date, asOfDate)
      
      if (daysPastDue < 0) aging.current += invoice.balance_due
      else if (daysPastDue <= 30) aging['1_30'] += invoice.balance_due
      else if (daysPastDue <= 60) aging['31_60'] += invoice.balance_due
      else if (daysPastDue <= 90) aging['61_90'] += invoice.balance_due
      else aging['over_90'] += invoice.balance_due
      
      return aging
    }, { current: 0, '1_30': 0, '31_60': 0, '61_90': 0, over_90: 0 })
  }
}
